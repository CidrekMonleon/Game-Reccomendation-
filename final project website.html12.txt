<script>
  const games = [
    { name: "Elden Ring", genre: "RPG", desc: "An open-world fantasy RPG full of mystery and challenge." },
    { name: "Valorant", genre: "Shooter", desc: "Fast-paced tactical shooter with unique agent abilities." },
    { name: "Stardew Valley", genre: "Simulation", desc: "A relaxing life and farm simulation game." },
    { name: "Hollow Knight", genre: "Adventure", desc: "A deep, atmospheric action-platformer." },
    { name: "Celeste", genre: "Platformer", desc: "Climb a mountain and overcome your fears." },
    { name: "Cyberpunk 2077", genre: "Action RPG", desc: "A story-driven open-world set in the future." },
    { name: "Overwatch 2", genre: "Shooter", desc: "Team-based hero shooter with colorful characters." }
  ];

  const loginPage = document.getElementById("loginPage");
  const homePage = document.getElementById("homePage");
  const userDisplay = document.getElementById("userDisplay");
  const gameList = document.getElementById("gameList");

  function login() {
    const username = document.getElementById("username").value.trim();
    const password = document.getElementById("password").value.trim();

    if (!username || !password) {
      alert("Please enter both username and password!");
      return;
    }

    localStorage.setItem("currentUser", username);
    showHome(username);
  }

  function logout() {
    localStorage.removeItem("currentUser");
    homePage.classList.add("hidden");
    loginPage.classList.remove("hidden");
  }

  function showHome(user) {
    loginPage.classList.add("hidden");
    homePage.classList.remove("hidden");
    userDisplay.textContent = user;

    displayGames(games);
  }

  function displayGames(gameArray) {
    gameList.innerHTML = "";
    if (gameArray.length === 0) {
      gameList.innerHTML = "<p>No games found. Try another search!</p>";
      return;
    }

    gameArray.forEach(g => {
      const card = document.createElement("div");
      card.className = "game-card";
      card.innerHTML = `
        <h3>${g.name}</h3>
        <p><b>Genre:</b> ${g.genre}</p>
        <p>${g.desc}</p>
        <!-- TODO: Add rating and favorite buttons -->
      `;
      gameList.appendChild(card);
    });
  }

  function searchGames() {
    const query = document.getElementById("searchInput").value.toLowerCase();
    const filtered = games.filter(g => g.name.toLowerCase().includes(query) || g.genre.toLowerCase().includes(query));
    displayGames(filtered);
  }

  window.onload = function() {
    const savedUser = localStorage.getItem("currentUser");
    if (savedUser) showHome(savedUser);
  };
</script>
